### Step 2: Calculate Pod Compatibility
```python
def calculate_pod_resonance(people_profiles, project_needs):
    """Find optimal pod composition for project"""
    
    # Required elements for this project type
    required_elements = project_needs['elements']  # e.g., [EARTH, WATER, FIRE]
    
    # Score each possible combination
    best_pods = []
    
    for combination in itertools.combinations(people_profiles, len(required_elements)):
        # Calculate average compatibility
        total_friction = 0
        
        for i, person1 in enumerate(combination):
            for person2 in combination[i+1:]:
                friction = elemental_mismatch_penalty(
                    person1['element'],
                    person2['element']
                )
                total_friction += friction
        
        avg_friction = total_friction / (len(combination) * (len(combination) - 1) / 2)
        
        # Check if combination covers required elements
        elements_covered = {p['element'] for p in combination}
        coverage = len(elements_covered & set(required_elements)) / len(required_elements)
        
        # Score = low friction + high coverage
        score = (1 - avg_friction) * coverage
        
        best_pods.append({
            'members': combination,
            'score': score,
            'avg_friction': avg_friction,
            'coverage': coverage
        })
    
    # Return top pods
    return sorted(best_pods, key=lambda x: x['score'], reverse=True)
```
